#include<iostream.h>
class A
{
public:
	virtual void fun(){cout<<"fun from a "<<endl;}
	virtual void func(){cout<<"func from a"<<endl;}

};
class B:public A
{
public:
	B(){func();}
	void test(){fun();}
	~B(){fun();}
};
class C:public B
{
public:
    
	void fun(){cout<<"fun from c"<<endl;}
	void func(){cout<<"func from c"<<endl;}
};
void main()
{
   C c;
   c.test();
    
}
输出结果：
  构造B时：   fun from A;
  C调用test： fun from c;
  析构B时：   fun from A;
